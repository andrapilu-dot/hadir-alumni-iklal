        // Helper: Convert timestamp "YYYY-MM-DD HH:MM" to "DD Bulan YYYY • HH:MM WIB"
        function formatIndonesianDate(timestampStr) {
            if (!timestampStr) return "";
            try {
                const parts = timestampStr.split(' ');
                const datePart = parts[0]; // "YYYY-MM-DD"
                const timePart = parts[1] || "00:00"; // "HH:MM"
                
                const dateParts = datePart.split('-');
                const year = dateParts[0];
                const monthNum = parseInt(dateParts[1]);
                const day = parseInt(dateParts[2]);
                
                const months = [
                    "Januari", "Februari", "Maret", "April", "Mei", "Juni",
                    "Juli", "Agustus", "September", "Oktober", "November", "Desember"
                ];
                const monthName = months[monthNum - 1] || "";
                
                return `${day} ${monthName} ${year} • ${timePart} WIB`;
            } catch (e) {
                return timestampStr; // Fallback
            }
        }

        // 2. Setup Real-time listener for Public Path (RULE 1 & RULE 2)
        function setupFirestoreListener() {
            if (!currentUser) return;

            // Strict collection path matching rule 1
            const collectionRef = collection(db, 'artifacts', appId, 'public', 'data', 'guests');

            // Listening to the collection in real-time (Rule 2: Fetch all, sort and filter in JS memory)
            onSnapshot(collectionRef, (snapshot) => {
                guests = [];
                snapshot.forEach((docSnap) => {
                    guests.push({
                        id: docSnap.id,
                        ...docSnap.data()
                    });
                });

                // Rule 2 Sorting (Sort by Timestamp Descending in memory)
                guests.sort((a, b) => {
                    const timeA = new Date(a.timestamp || 0);
                    const timeB = new Date(b.timestamp || 0);
                    return timeB - timeA;
                });

                // Update local storage backup
                localStorage.setItem('alumni_guests', JSON.stringify(guests));
                updateDashboard();
            }, (error) => {
                console.error("Firestore onSnapshot error: ", error);
                switchToLocalMode();
            });
        }

        // Add guest to Firestore Database (RULE 1 & RULE 3 check) or Local backup
        async function handleAddGuest(e) {
            e.preventDefault();

            const fullName = fullNameInput.value.trim();
            const genderNode = document.querySelector('input[name="gender"]:checked');
            const gender = genderNode ? genderNode.value : '';
            const level = document.getElementById('level').value;
            const gradYear = parseInt(document.getElementById('gradYear').value);
            const phone = document.getElementById('phone').value.trim();
            const korda = document.getElementById('korda').value;
            const domicile = document.getElementById('domicile').value.trim();
            const occupation = document.getElementById('occupation').value.trim() || '-';

            // Simple validation check
            if (!fullName || !gender || !level || !gradYear || !phone || !korda || !domicile) {
                showToast("Peringatan", "Mohon isi semua field wajib (*)", true);
                fullNameInput.focus();
                return;
            }

            // ANTI-DUPLICATE CHECK: Prevent identical Name and Level combination
            const isDuplicate = guests.some(g => 
                g.id !== editingGuestId &&
                g.fullName.trim().toLowerCase() === fullName.toLowerCase() && 
                g.level === level
            );

            if (isDuplicate) {
                showToast(
                    "Pendaftaran Gagal", 
                    `Alumni bernama "${fullName}" sudah terdaftar pada jenjang yang sama.`, 
                    true
                );
                fullNameInput.focus();
                fullNameInput.select(); 
                return;
            }

            // Timestamp decision: keep original timestamp on edit mode, generate new on create mode
            let timestamp;
            if (editingGuestId && editingTimestamp) {
                timestamp = editingTimestamp;
            } else {
                const now = new Date();
                timestamp = now.getFullYear() + '-' + 
                    String(now.getMonth() + 1).padStart(2, '0') + '-' + 
                    String(now.getDate()).padStart(2, '0') + ' ' + 
                    String(now.getHours()).padStart(2, '0') + ':' + 
                    String(now.getMinutes()).padStart(2, '0');
            }

            // Keep existing WA sent status on edit, or initialize as false for new records (since auto-send is removed)
            let waSentValue = false;
            if (editingGuestId) {
                const existing = guests.find(g => g.id === editingGuestId);
                waSentValue = existing ? (existing.waSent || false) : false;
            }

            const guestData = {
                fullName,
                gender,
                level,
                gradYear,
                phone,
                korda,
                domicile,
                occupation,
                timestamp,
                waSent: waSentValue
            };

            if (editingGuestId) {
                // --- UPDATE REGISTERED GUEST MODE ---
                const updatedLocalGuests = guests.map(g => {
                    if (g.id === editingGuestId) {
                        return { id: editingGuestId, ...guestData };
                    }
                    return g;
                });
                localStorage.setItem('alumni_guests', JSON.stringify(updatedLocalGuests));

                if (isCloudMode && currentUser) {
                    try {
                        const docRef = doc(db, 'artifacts', appId, 'public', 'data', 'guests', editingGuestId);
                        await setDoc(docRef, guestData);
                    } catch (err) {
                        console.error("Cloud write failed, continuing with local data:", err);
                        guests = updatedLocalGuests;
                        updateDashboard();
                    }
                } else {
                    guests = updatedLocalGuests;
                    updateDashboard();
                }

                showToast("Sukses Diperbarui!", `Data atas nama "${fullName}" berhasil diperbarui.`);
                cancelEdit(); 
            } else {
                // --- CREATE NEW GUEST MODE ---
                const guestId = 'alumni-' + Date.now();
                const updatedLocalGuests = [...guests];
                updatedLocalGuests.unshift({ id: guestId, ...guestData });
                localStorage.setItem('alumni_guests', JSON.stringify(updatedLocalGuests));

                if (isCloudMode && currentUser) {
                    try {
                        // Write to Firebase Cloud
                        const colRef = collection(db, 'artifacts', appId, 'public', 'data', 'guests');
                        await addDoc(colRef, guestData);
                    } catch (err) {
                        console.error("Cloud write failed, continuing with local data:", err);
                        guests = updatedLocalGuests;
                        updateDashboard();
                    }
                } else {
                    guests = updatedLocalGuests;
                    updateDashboard();
                }

                // Reset UI & autofocus
                form.reset();
                showToast("Sukses Disimpan!", `Atas nama "${fullName}" berhasil didaftarkan secara aman.`);
                fullNameInput.focus();

        }

        // Delete guest from Cloud and Local
        async function deleteGuest(id, name) {
            openConfirmModal(
                "Hapus Data Alumni", 
                `Apakah Anda yakin ingin menghapus data alumni "${name}"?`, 
                async () => {
                    const updatedLocalGuests = guests.filter(g => g.id !== id);
                    localStorage.setItem('alumni_guests', JSON.stringify(updatedLocalGuests));

                    if (isCloudMode && currentUser) {
                        try {
                            const docRef = doc(db, 'artifacts', appId, 'public', 'data', 'guests', id);
                            await deleteDoc(docRef);
                        } catch (err) {
                            console.error("Cloud delete failed, continuing with local data:", err);
                            guests = updatedLocalGuests;
                            updateDashboard();
                        }
                    } else {
                        guests = updatedLocalGuests;
                        updateDashboard();
                    }
                    showToast("Terhapus", `Data alumni "${name}" berhasil dihapus.`);
                    fullNameInput.focus();
                }
            );
        }

        // Clear all documents from the Database
        async function handleClearDatabase() {
            openConfirmModal(
                "Kosongkan Database", 
                "PERINGATAN: Semua data pendaftaran alumni akan dihapus secara permanen! Tindakan ini tidak dapat dibatalkan.", 
                async () => {
                    localStorage.setItem('alumni_guests', JSON.stringify([]));

                    if (isCloudMode && currentUser) {
                        try {
                            const colRef = collection(db, 'artifacts', appId, 'public', 'data', 'guests');
                            const querySnap = await getDocs(colRef);
                            
                            const deletePromises = [];
                            querySnap.forEach((docSnap) => {
                                const docRef = doc(db, 'artifacts', appId, 'public', 'data', 'guests', docSnap.id);
                                deletePromises.push(deleteDoc(docRef));
                            });

                            await Promise.all(deletePromises);
                        } catch (err) {
                            console.error("Cloud clear failed:", err);
                        }
                    }
                    
                    guests = [];
                    updateDashboard();
                    showToast("Dikosongkan", "Semua data berhasil dibersihkan.", true);
                    fullNameInput.focus();
                }
            );
        }

        // Seed mock data for demo presentation
        async function seedMockData() {
            const tempGuests = [...guests];
            initialMockData.forEach(data => {
                tempGuests.unshift({
                    id: 'mock-' + Math.random(),
                    ...data
                });
            });

            localStorage.setItem('alumni_guests', JSON.stringify(tempGuests));

            if (isCloudMode && currentUser) {
                try {
                    const colRef = collection(db, 'artifacts', appId, 'public', 'data', 'guests');
                    const addPromises = initialMockData.map(data => addDoc(colRef, data));
                    await Promise.all(addPromises);
                } catch (err) {
                    console.error("Cloud seeding failed:", err);
                }
            } else {
                guests = tempGuests;
                updateDashboard();
            }

            showToast("Simulasi Berhasil", "Berhasil memuat data contoh!");
            fullNameInput.focus();
        }

        // Export Data to Real Excel (.xlsx) file
        function exportToExcel() {
            if (guests.length === 0) {
                showToast("Data Kosong", "Tidak ada data tamu untuk diekspor saat ini.", true);
                fullNameInput.focus();
                return;
            }

            const searchValue = searchInput.value.toLowerCase().trim();
            const selectedLevel = filterLevel.value;
            const selectedKorda = filterKorda.value;

            const exportDataList = guests.filter(g => {
                const matchesLevel = (selectedLevel === "ALL" || g.level === selectedLevel);
                const matchesKorda = (selectedKorda === "ALL" || g.korda === selectedKorda);
                const matchesSearch = 
                    g.fullName.toLowerCase().includes(searchValue) ||
                    (g.korda && g.korda.toLowerCase().includes(searchValue)) ||
                    g.domicile.toLowerCase().includes(searchValue) ||
                    g.occupation.toLowerCase().includes(searchValue) ||
                    g.phone.includes(searchValue);

                return matchesLevel && matchesKorda && matchesSearch;
            });

            if (exportDataList.length === 0) {
                showToast("Tidak Ada Cocok", "Tidak ada baris data yang cocok dengan filter.", true);
                fullNameInput.focus();
                return;
            }

            const formattedData = exportDataList.map((g, index) => ({
                'No': index + 1,
                'Nama Lengkap': g.fullName,
                'Jenis Kelamin': g.gender === 'L' ? 'Laki-laki (L)' : (g.gender === 'P' ? 'Perempuan (P)' : '-'),
                'Jenjang Pendidikan': g.level,
                'Tahun Lulus (Angkatan)': g.gradYear,
                'No WhatsApp': g.phone,
                'Konfirmasi WA': g.waSent ? 'Terkirim (✓)' : 'Belum Dikirim',
                'Korda (Koordinator Daerah)': g.korda || 'Lainnya',
                'Domisili Saat Ini': g.domicile,
                'Pekerjaan / Aktivitas': g.occupation,
                'Waktu Kehadiran': g.timestamp
            }));

            const worksheet = XLSX.utils.json_to_sheet(formattedData);

            const colWidths = [
                { wch: 6 }, { wch: 30 }, { wch: 18 }, { wch: 20 }, { wch: 22 },
                { wch: 20 }, { wch: 18 }, { wch: 25 }, { wch: 25 }, { wch: 30 }, { wch: 20 }
            ];
            worksheet['!cols'] = colWidths;

            const workbook = XLSX.utils.book_new();
            XLSX.utils.book_append_sheet(workbook, worksheet, "Daftar Kehadiran Alumni");

            const today = new Date();
            const dateString = today.getFullYear() + '-' + 
                String(today.getMonth() + 1).padStart(2, '0') + '-' + 
                String(today.getDate()).padStart(2, '0');
            
            let filterSuffix = "";
            if (selectedLevel !== "ALL") filterSuffix += `_${selectedLevel}`;
            if (selectedKorda !== "ALL") filterSuffix += `_Korda_${selectedKorda}`;

            const fileName = `Daftar_Hadir_Alumni_IKLAL_${dateString}${filterSuffix}.xlsx`;
            XLSX.writeFile(workbook, fileName);

            showToast("Sukses Ekspor Excel", `Berhasil mengunduh ${exportDataList.length} data ke Excel!`);
            fullNameInput.focus();
        }

        // Helper string sanitization functions
        function escapeHTML(str) {
            if (!str) return '';
            return str
                .replace(/&/g, '&amp;')
                .replace(/</g, '&lt;')
                .replace(/>/g, '&gt;')
                .replace(/"/g, '&quot;')
                .replace(/'/g, '&#039;');
        }

        function formatWhatsAppNumber(phoneNum) {
            let cleared = phoneNum.replace(/[^0-9]/g, '');
            if (cleared.startsWith('0')) {
                cleared = '62' + cleared.slice(1);
            }
            return cleared;
        }
